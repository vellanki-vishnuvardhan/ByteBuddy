<%
    String jdbcUrl = "jdbc:mysql://localhost:3306/first";
    String username = "root";
    String password = "renuka";

    // Function to get the username from the session
    String getUsername() {
        HttpSession session = request.getSession();
        return (String) session.getAttribute("name"); // Change to the attribute used in your signup page
    }

    // Function to get the quiz score for a specific quiz type
    int getQuizScore(String quizType) {
        int score = 0;

        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            // Query to retrieve the score based on username and quiz type
            String selectQuery = "SELECT score FROM user_scores WHERE username=? AND quiz_type=?";
            try (Connection connection = DriverManager.getConnection(jdbcUrl, username, password);
                 PreparedStatement preparedStatement = connection.prepareStatement(selectQuery)) {

                // Get the current username from the session
                String currentUser = getUsername();

                // Set parameters in the prepared statement
                preparedStatement.setString(1, currentUser);
                preparedStatement.setString(2, quizType);

                try (ResultSet resultSet = preparedStatement.executeQuery()) {
                    // Check if a result is found and retrieve the score
                    if (resultSet.next()) {
                        score = resultSet.getInt("score");
                    }
                }
            }
        } catch (ClassNotFoundException | SQLException e) {
            e.printStackTrace();
            // Handle the exception
        }

        return score;
    }
%>
